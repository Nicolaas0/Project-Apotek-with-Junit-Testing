     private void search (String type, String query){ //untuk search
         try {
             if (query.isEmpty()){
                 this.rs = controller.get();
             }
             else if (type.equals("id") && !query.isEmpty()){
                 this.rs = controller.showById(query);
             }
             else if (type.equals("name") && !query.isEmpty()){
                 this.rs = controller.showByName(query);
             }
             
             loadTable(this.rs);
         } catch (Exception e) {
             System.out.println(e.getMessage());
         }
     }

    public void loadTable(ResultSet rs){
        tb_product1.setModel(DbUtils.resultSetToTableModel(rs));
        
        TableColumnModel tcm = tb_product1.getColumnModel();
        tcm.getColumn(0).setHeaderValue("Product ID");
        tcm.getColumn(1).setHeaderValue("Name");
        tcm.getColumn(2).setHeaderValue("Type");
        tcm.getColumn(3).setHeaderValue("Stock");
        tcm.getColumn(4).setHeaderValue("Price");
        tcm.getColumn(5).setHeaderValue("Expired_Date");
    }

    private void btn_submitActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btn_submitActionPerformed
        // TODO add your handling code here:                                      //SUBMIT BUTTON
        try {            
        String name = tf_name.getText();
        String address = tf_address.getText();
        String id_obat = tf_idobat.getText();
        String product_name = tf_productname.getText();
        String totalq = tf_totalq.getText();
        String totalp = tf_totalp.getText();
        
        model.setName(name);
        model.setAddress(address);
        model.setId_obat(id_obat);
        model.setProduct_Name(product_name);
        model.setTotalq(totalq);
        model.setTotalp(totalp);
        
        boolean result = controller.submit(model); 
         if(result){
            JOptionPane.showMessageDialog(null,"Pesanan" + product_name + "atas nama"
            + name +"dengan total harga" + totalp + "akan di kirimkan ke" + address
            + ", Terimakasih sudah menggunakan layanan aplikasi kami!");
        }
        else{
            JOptionPane.showMessageDialog(null, "Gagal Menlakukan transaksi");
        }
        Clear();
        getAllData();
        } catch (Exception ex) {
            System.err.println(ex.getMessage());
        }
    }//GEN-LAST:event_btn_submitActionPerformed
