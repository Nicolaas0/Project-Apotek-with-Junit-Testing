    private void btn_sign_inActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btn_sign_inActionPerformed
        // TODO add your handling code here
        try {
            String username = tf_username.getText();
            String password = tf_password.getText();

            MainController mc = new MainController();
            MainModel model = new MainModel();
            model.setUsername(username);
            model.setPassword(password);
            mc.login(model, password);

            String result = mc.login(model, password);
            switch (result) {
                case "A":
                    JOptionPane.showMessageDialog(null, "Welcome back, Admin");
                    Adminframe adminmf = new Adminframe();
                    this.setVisible(false);
                    adminmf.setVisible(true);
                    dispose();
                    break;
                case "U":
                    JOptionPane.showMessageDialog(null, "You Are not an admin");
                    Mainframe usermf = new Mainframe();
                    this.setVisible(false);
                    usermf.setVisible(true);
                    dispose();
                    break;
                default:
                    JOptionPane.showMessageDialog(null, "Login Failed");
                    break;
            }
        } catch (Exception e) {
            System.err.println(e.getMessage());
        }
    }//GEN-LAST:event_btn_sign_inActionPerformed
