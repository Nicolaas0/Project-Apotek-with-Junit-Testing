    private void btn_submitinpActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btn_submitinpActionPerformed
        // TODO add your handling code here:                //UNTUK SUMBIT
        try {
            String name = tf_nameinput.getText();
            String price = tf_priceinput.getText();
            String stock = tf_stock.getText();
            String type = cb_type.getSelectedItem().toString();
            String date = tf_date.getText();
            
            model.setName(name);
            model.setPrice(price);
            model.setStock(stock);
            model.setType(type);
            model.setExpired_date(date);
            
            Boolean result  = controller.create(model);
            if (result) {
                JOptionPane.showMessageDialog(null, "Berhasil Menambahkan Data");
            } else {
                JOptionPane.showMessageDialog(null, "Gagal Menambahkan Data");
            }
            getAllData();
            clear();
        } catch (Exception ex) {
            System.out.println(ex.getMessage());
        }
        
    }//GEN-LAST:event_btn_submitinpActionPerformed

    private void btn_updateActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btn_updateActionPerformed
        // TODO add your handling code here:               //UNTUK UPDATE
         try {
            String name = tf_nameinput.getText();
            String type = cb_type.getSelectedItem().toString();
            String stock = tf_stock.getText();
            String price = tf_priceinput.getText();
            String date = tf_date.getText();
            
            model.setName(name);
            model.setPrice(price);
            model.setStock(stock);
            model.setType(type);
            model.setExpired_date(date);
            
            Boolean result = controller.update(product_id,model);
            if (result) {
                JOptionPane.showMessageDialog(null, "Berhasil Mengubah Data");
            } else {
                JOptionPane.showMessageDialog(null, "Gagal Mengubah Data" );
            }
            
            clear();
            getAllData();
        } catch (Exception ex) {
            System.err.println(ex.getMessage());
        }
    }//GEN-LAST:event_btn_updateActionPerformed
